```astro
---
import type { MarkdownLayoutProps } from "astro";
import Layout from "./Layout.astro";

type Props = MarkdownLayoutProps<{
    title: string;
    author: string
    updatedAt: string;
    tag: string[];
}>

const { frontmatter: blog } = Astro.props
---

<Layout title={`${blog} | timtim.tech`}>
    <div class="header">
        <h1>{blog.title}</h1>
        <p>{blog.author}</p>
        <p>{blog.updatedAt}</p>
        <ul>
            {blog.tag.map((tag) => (
                <li>{tag}</li>
            ))}
        </ul>
    </div>
    <div class="contents">
        <slot />
    </div>
</Layout>
```
